56 static inline void switchToMayaCamera(NirtcppDevice* device)
57 {
58#if 1
59 return;
60#else
61 if (!device) return;
62
63 scene::ICameraSceneNode* camera = device->getSceneManager()->getActiveCamera();
64 if (!camera || camera->getID() == 54321) return;
65
66
67 core::vector3df target = camera->getTarget() - camera->getPosition();
68 core::vector3df relativeRotation = target.getHorizontalAngle();
69
70 scene::ICameraSceneNode* maya = device->getSceneManager()->addCameraSceneNodeMaya(
71 0, -1500, 1000, 1500,
72 54321,
73 target.getLength(),
74 true,
75 relativeRotation.X + 90, relativeRotation.Y
76 );
77 if (maya)
78 {
79 maya->setNearValue(camera->getNearValue());
80 maya->setFarValue(camera->getFarValue());
81 }
82
83 device->getCursorControl()->setVisible(true);
84 device->setResizable(true);
85#endif
86 }
