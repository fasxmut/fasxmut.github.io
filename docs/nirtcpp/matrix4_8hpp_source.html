926 template <class T>
927 inline core::vector3d<T> CMatrix4<T>::getRotationDegrees(const vector3d<T>& scale_) const
928 {
929 const CMatrix4<T> &mat = *this;
930 const core::vector3d<f64> scale(core::iszero(scale_.X) ? FLT_MAX : scale_.X , core::iszero(scale_.Y) ? FLT_MAX : scale_.Y, core::iszero(scale_.Z) ? FLT_MAX : scale_.Z);
931 const core::vector3d<f64> invScale(core::reciprocal(scale.X),core::reciprocal(scale.Y),core::reciprocal(scale.Z));
932
933 f64 Y = -asin(core::clamp(mat[2]*invScale.X, -1.0, 1.0));
934 const f64 C = cos(Y);
935 Y *= RADTODEG64;
936
937 f64 rotx, roty, X, Z;
938
939 if (!core::iszero((T)C))
940 {
941 const f64 invC = core::reciprocal(C);
942 rotx = mat[10] * invC * invScale.Z;
943 roty = mat[6] * invC * invScale.Y;
944 X = atan2( roty, rotx ) * RADTODEG64;
945 rotx = mat[0] * invC * invScale.X;
946 roty = mat[1] * invC * invScale.X;
947 Z = atan2( roty, rotx ) * RADTODEG64;
948 }
949 else
950 {
951 X = 0.0;
952 rotx = mat[5] * invScale.Y;
953 roty = -mat[4] * invScale.Y;
954 Z = atan2( roty, rotx ) * RADTODEG64;
955 }
956
957 // fix values that get below zero
958 if (X < 0.0) X += 360.0;
959 if (Y < 0.0) Y += 360.0;
960 if (Z < 0.0) Z += 360.0;
961
962 return vector3d<T>((T)X,(T)Y,(T)Z);
963 }

966 template <class T>
967 inline core::vector3d<T> CMatrix4<T>::getRotationDegrees() const
968 {
969 // Note: Using getScale() here make it look like it could do matrix decomposition.
970 // It can't! It works (or should work) as long as rotation doesn't flip the handedness
971 // aka scale swapping 1 or 3 axes. (I think we could catch that as well by comparing
972 // crossproduct of first 2 axes to direction of third axis, but TODO)
973 // And maybe it should also offer the solution for the simple calculation
974 // without regarding scaling as Nirtcpp did before 1.7
975 core::vector3d<T> scale(getScale());
976
977 // We assume the matrix uses rotations instead of negative scaling 2 axes.
978 // Otherwise it fails even for some simple cases, like rotating around
979 // 2 axes by 180° which getScale thinks is a negative scaling.
980 if (scale.Y<0 && scale.Z<0)
981 {
982 scale.Y =-scale.Y;
983 scale.Z =-scale.Z;
984 }
985 else if (scale.X<0 && scale.Z<0)
986 {
987 scale.X =-scale.X;
988 scale.Z =-scale.Z;
989 }
990 else if (scale.X<0 && scale.Y<0)
991 {
992 scale.X =-scale.X;
993 scale.Y =-scale.Y;
994 }
995
996 return getRotationDegrees(scale);
997 }
