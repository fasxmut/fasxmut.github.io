437 {
438 public:
439 T c;
440 xmlChar() {}
441 xmlChar(char in) : c(static_cast<T>(in)) {}
442 xmlChar(wchar_t in) : c(static_cast<T>(in)) {}
443#if defined(__BORLANDC__)
444 // Note - removing explicit for Borland was to get it to even compile.
445 // There haven't been any kind of tests for that besides that.
446 xmlChar(unsigned char in) : c(static_cast<T>(in)) {}
447 xmlChar(unsigned short in) : c(static_cast<T>(in)) {}
448 xmlChar(unsigned int in) : c(static_cast<T>(in)) {}
449 xmlChar(unsigned long in) : c(static_cast<T>(in)) {}
450#else
451 explicit xmlChar(unsigned char in) : c(static_cast<T>(in)) {}
452
453#if !defined(_MSC_VER) || defined(_NATIVE_WCHAR_T_DEFINED) // VS compiling without native wchar_t can't have it
454 explicit xmlChar(unsigned short in) : c(static_cast<T>(in)) {}
455#endif
456 explicit xmlChar(unsigned int in) : c(static_cast<T>(in)) {}
457 explicit xmlChar(unsigned long in) : c(static_cast<T>(in)) {}
458#endif
459 operator T() const { return c; }
460 void operator=(int t) { c=static_cast<T>(t); }
461 };
