kcpp::endl

kcpp::endl is an executor which outputs a newline character and flush the output stream.

kcpp::endl is similar to std::endl.

include 

#include <kcpp/io.hpp>

c++ example 

#include <kcpp/io.hpp>

int main()
{
	kcpp::kout << 23 << kcpp::endl << 45 << kcpp::endl;
/*
23
45
*/
}
