kjcpp::sleep

kjcpp::sleep - kjcpp

kjcpp::sleep is the kjcpp c++ object of kjcpp::sleep_class.

kjcpp::sleep

Sleep for secs seconds, floating value is accepted : 

kjcpp::sleep(secs);

Sleep for dura seconds, chrono-literal-time is accepted: 

kjcpp::sleep(dura);

Sleep for weeks, days, hours, minutes, seconds, and milliseconds, floating value is accepted, chrono-literal value is not accepted : 

kjcpp::sleep(weeks, days, hours, minutes, seconds, milliseconds);

kjcpp::sleep.msleep

Sleep for milli_seconds, floating value: 

kjcpp::sleep.msleep(milli_seconds);

kjcpp::sleep.usleep

Sleep for micro_seconds, floating value: 

kjcpp::sleep(micro_seconds);

kjcpp::sleep.nsleep

Sleep for nano_seconds, floating value: 

kjcpp::nsleep(nano_seconds);

kjcpp::sleep.minsleep

Sleep for minutes, floating value: 

kjcpp::sleep.minsleep(minutes);

kjcpp::sleep.hsleep

Sleep for hours, floating value: 

kjcpp::sleep(hours);

kjcpp::sleep.dsleep

Sleep for days, floating value: 

kjcpp::sleep.dsleep(days);

kjcpp::sleep.wsleep

Sleep for weeks, floating value: 

kjcpp::sleep.wsleep(weeks);
