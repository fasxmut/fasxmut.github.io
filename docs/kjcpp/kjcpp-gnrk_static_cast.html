kjcpp::gnrk_static_cast

kjcpp::gnrk_static_cast - Kjcpp Generic Static Cast.

kjcpp::gnrk_static_cast tries to static_cast types as more as possible: kjcpp::meric_static_cast, simple-constructible-with cast, and static_cast .

kjcpp::gnrk_static_cast rules over three casts:

  • kjcpp::meric_static_cast
  • simple-constructible-with cast
  • static_cast

kjcpp::gnrk_static_cast c++ example 

#include <kjcpp/core.hpp>

int main()
{
	constexpr kjcpp::ix32 a = 123;
	constexpr auto b = kjcpp::gnrk_static_cast<kjcpp::fx64>(a);	// It calls kjcpp::meric_static_cast internally
	kjcpp::print(a, b);	// 123 123.0

	constexpr float c = 3.4;
	constexpr auto d = kjcpp::gnrk_static_cast<int>(c);	// It calls static_cast internally
	kjcpp::print(c, d);	// 3.4 3

	class cat_box
	{
	private:
		kjcpp::ix32 value;
	public:
		constexpr cat_box(const kjcpp::ix32 & value): value{value}
		{
		}
		constexpr auto operator()() const
		{
			return value;
		}
	};
	constexpr kjcpp::ix32 e = 234;
	constexpr auto f = kjcpp::gnrk_static_cast<cat_box>(e);	// It calls target_type{value__} internally
	kjcpp::print(e, f());	// 234 234

	kjcpp::ix32 g = 345;
	decltype(auto) h = kjcpp::gnrk_static_cast<kjcpp::ix32 &>(g);
	++h;
	kjcpp::print(g, h);	// 346 346
}
