Example 

#include <kjcpp/core.hpp>

int main() {
	kjcpp::print(
		"Your Name is:",
		kjcpp::readline("Please Input Your Name: ").read(),
		"\nHow are you."
	);
}
Result: 
Please Input Your Name: Fas Xmut
Your Name is: Fas Xmut
How are you.
