kjcpp::nan_v, kjcpp::is_nan

kjcpp::nan_v

kjcpp::nan_v - gets the NaN value of specific real type.

kjcpp::nan_v can be applied to floating types only. 

template <kjcpp::floating_meric type1>
constexpr inline type1 nan_v = kjcpp::nan_v_impl<type1>::value;

kjcpp::is_nan

kjcpp::is_nan - used to judge if a floating value is NaN or not.

kjcpp::is_nan can be applied to floating types only. 

template <kjcpp::floating_meric float_type>
constexpr inline bool is_nan(const float_type & value);

c++ example 

#include <kjcpp/core.hpp>
#include <kjcpp/math.hpp>

int main()
{
	kjcpp::fx32 a = kjcpp::nan_v<kjcpp::fx32>;
	kjcpp::fx32 b = 123.321f;
	kjcpp::fx32 c = kjcpp::sqrt(-1.0f);

	kjcpp::print(kjcpp::is_nan(a), kjcpp::is_nan(b), kjcpp::is_nan(c)); // true false true

	kjcpp::print(a, b, c); // nan 123.321 nan
}
