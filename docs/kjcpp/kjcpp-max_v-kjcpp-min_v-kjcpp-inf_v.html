kjcpp::max_v, kjcpp::min_v, kjcpp::inf_v

kjcpp::max_v

kjcpp::max_v - gets the maxium value of specific real type.

kjcpp::max_v can be applied to integral or floating point types. 

template <kjcpp::real_meric type1>
constexpr inline type1 max_v = kjcpp::max_v_impl<type1>::value;

kjcpp::min_v

kjcpp::min_v - gets the minimum value of specific type.

kjcpp::min_v can be applied to integral or floating point types. 

template <kjcpp::real_meric type1>
constexpr inline type1 min_v = kjcpp::min_v_impl<type1>::value;

kjcpp::inf_v

kjcpp::inf_v - gets the infinite of specific floating type.

kjcpp::inf_v can only be applied to floating point types. 

template <kjcpp::floating_meric type1>
constexpr inline type1 inf_v = kjcpp::inf_v_impl<type1>::value;

c++ example 

#include <kjcpp/core.hpp>

int main()
{
	kjcpp::print(
		kjcpp::max_v<kjcpp::u16>,
		kjcpp::min_v<kjcpp::u16>
	);

	kjcpp::print(
		kjcpp::max_v<kjcpp::f32>,
		kjcpp::min_v<kjcpp::f32>,
		kjcpp::inf_v<kjcpp::f32>
	);
}
