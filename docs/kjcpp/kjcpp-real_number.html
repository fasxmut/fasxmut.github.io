kjcpp::real_number

kjcpp::real_number is a concept of kjcpp that requires the types must satisfy integral or floating point requirements.

It means real number in math.

Example

c++ example 

#include <kjcpp/core.hpp>

template <kjcpp::real_number type_xta>
class my_data
{
};

int main()
{
	my_data<kjcpp::i32> d1;	// OK
	//my_data<kjcpp::ix32> d2;	// ERROR: kjcpp::ix32 is class type of kjcpp::i32
	static_assert(kjcpp::real_number<kjcpp::i32>);
	static_assert(! kjcpp::real_number<kjcpp::ix32>);
	kjcpp::real_number auto a = 345;	// a is int
	kjcpp::real_number auto b = 3.45;	// b is double
}
