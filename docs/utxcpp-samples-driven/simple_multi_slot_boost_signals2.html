Simple Multi Slot - boost::signals2

A signal connects multiple slot sources.

The Boost.Signals2 library is an implementation of a managed signals and slots system.

This program shows how utxcpp work with boost::signals2.

(utx::ix32) (utx::print) (utx::string_literals) (.cat - utx::string)

#include <utxcpp/core.hpp>
#include <boost/signals2.hpp>

namespace signals = boost::signals2;
using namespace utx::string_literals;
using namespace utx::uct_literals;

class device
{
public:
	void operator()()
	{
		utx::print("Device is booted.");
	}
};

template <utx::ix32 __id>
class memory
{
private:
	utx::ix32 id = __id;
public:
	void operator()()
	{
		utx::print("Memory #"_us.cat(id), "is filled.");
	}
};

auto task = []
{
	utx::print("Device is working.");
};

int main()
{
	signals::signal<void(void)> sigv;
	sigv.connect(device{});
	sigv.connect(memory<1_uct>{});
	sigv.connect(memory<2_uct>{});
	sigv.connect(task);
	sigv();
}

output: 

Device is booted.
Memory #1 is filled.
Memory #2 is filled.
Device is working.
