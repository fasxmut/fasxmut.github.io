std::bit_floor, std::bit_ceil

std::bit_floor calculates the largest integral power of two that is not greater than x. If x is zero, returns zero.

std::bit_ceil calculates the smallest integral power of two that is not less than x.

std::bit_floor and std::bit_ceil are c++20 library features.

(utx::ux32) (utx::ux16) (utx::print)

#include <bit>
#include <utxcpp/core.hpp>

int main()
{
	constexpr utx::ux32 x1 = 1234;
	constexpr utx::ux16 x2 = 2049;

	constexpr utx::ux32 x3 = std::bit_floor(x1());
	constexpr utx::ux32 x4 = std::bit_ceil(x1());

	constexpr utx::ux16 x5 = std::bit_floor(x2());
	constexpr utx::ux16 x6 = std::bit_ceil(x2());

	utx::print(x3, x4, x5, x6); // 1024 2048 2048 4096
}
