Related Types - boost::signals2 Utxcpp

Related Types

namespace 

namespace signals = boost::signals2;

Signal Type 

using signal_t = signals::signal<return_type(arg_type)>;

Slot Type 

using slot_t = signal_t::slot_type;

Slot Type object is initialized with source function object or lambda function, by sig.connect(slot_function) . 

slot_t slot1 = function1;
sig.connect(function1); // or: sig.connect(slot1);

Connection Type 

using connection_t = signals::connection;

Connection Type is the sig.connect(slot) returned type. 

connection_t conn1 = sig.connect(slot1);

Result Type

Result type is boost::optional<data_type>, it is returned by calling the signal: 

boost::optional<utx::fx32> result = sig(1.234f);
utx::print(*result); // print the result

Calling signal sig() will return the last slot returned result, if it has multiple slots.

#include <utxcpp/core.hpp>
#include <utxcpp/math.hpp>
#include <boost/signals2.hpp>

namespace signals = boost::signals2;

auto transform = [] (utx::fx32 x)
{
	return x*x;
};

int main()
{
	using signal_t = signals::signal<utx::fx32(utx::fx32)>;
	using slot_t = signal_t::slot_type;
	using connection_t = signals::connection;

	slot_t slot1 = transform;
	slot_t slot2 = utx::exp<utx::fx32>;

	signal_t sig;

	connection_t conn1 = sig.connect(slot1);
	connection_t conn2 = sig.connect(slot2);

	boost::optional<utx::fx32> result = sig(2.5f); // Return the last slot returned result.

	conn1.disconnect();
	conn2.disconnect();

	utx::print("*result =>", *result);
}
