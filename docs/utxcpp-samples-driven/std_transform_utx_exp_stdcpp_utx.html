std::transform and utx::exp, utx::cos, utx::sin

std::transform is a c++ STL algorithm to transform a range with given algorithm, every element of the range is transformed with the same algorithm.

std::ranges::transform, the c++20 ranges of std::transform.

utx::exp, utx::cos, utx::sin are provided by utxcpp library.

(utx::exp) (utx::cos) (utx::sin) (utx::fx32) (utx::print)

#include <algorithm>
#include <vector>
#include <numeric>
#include <utxcpp/math.hpp>
#include <utxcpp/core.hpp>

int main()
{
	std::vector<utx::fx32> vector(13); // floating vector
	std::iota(vector.begin(), vector.end(), 1.0f);

	std::vector<utx::fx32> output(vector.size());
	std::transform(
		vector.begin(),
		vector.end(),
		output.begin(),
		[] (const utx::fx32 & value)
		{
			utx::fx32 tmp = utx::exp(value);
			tmp = utx::cos(tmp);
			return tmp;
		}
	);

	std::vector<utx::fx32> output2(output.size());
	std::ranges::transform(
		output,
		output2.begin(),
		[] (const utx::fx32 & value)
		{
			utx::fx32 tmp = utx::exp(value);
			tmp = utx::sin(tmp);
			return tmp;
		}
	);

	utx::print(std::setprecision(6), vector, output, output2);
}
