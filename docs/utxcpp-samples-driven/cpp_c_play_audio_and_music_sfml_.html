#include <SFML/Audio.hpp>
#include <utxcpp/core.hpp>
#include <utxcpp/thread.hpp>

using namespace utx::string_literals;

int main(int argc, char * argv[])
try
{
	if (argc != 2)
		throw std::runtime_error{"Usage: "_us.cat(argv[0], " <ogg audio file>")};
	const utx::string filename = argv[1];

	sf::Music music;
	bool status = music.openFromFile(filename);
	utx::rt_assert(status, "Music is opened");
	music.play();

	utx::sleep(0.6);

	sf::SoundBuffer buffer;
	status = buffer.loadFromFile(filename);
	utx::rt_assert(status, "Sound buffer is loaded");

	sf::Sound snd;
	snd.setBuffer(buffer);
	snd.play();
	utx::sleep(0.12);
	sf::Sound snd2;
	snd2.setBuffer(buffer);
	snd2.play();

	while (music.getStatus() == sf::Music::Playing || snd.getStatus() == sf::Sound::Playing || snd2.getStatus() == sf::Sound::Playing)
	{
		utx::sleep(0.3);
	}
	utx::print("Music and sound are finished.");
}
catch (const std::exception & err)
{
	utx::print("================================================================================");
	utx::printe("std::exception:", err.what());
	return 1;
}
