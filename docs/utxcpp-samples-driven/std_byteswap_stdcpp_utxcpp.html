std::byteswap

std::byteswap: Reverses the bytes in the given integer value n . It is c++23 library feature.

(std::byteswap) (utx::ux32) (utx::ux16) (utx::print)

#include <bit>
#include <utxcpp/core.hpp>

int main()
{
	{
		constexpr utx::u32 x1 = 0x12345678;
		constexpr utx::u16 x2 = 0x1234;
		constexpr utx::u32 y1 = std::byteswap(x1); // OK
		constexpr utx::u32 y2 = std::byteswap(x2); // OK
		utx::print(std::hex, std::showbase, y1, y2); // 0x78563412 0x3412
	}
	{
		constexpr utx::ux32 x1 = 0x12345678;
		constexpr utx::ux16 x2 = 0x1234;
		
		//constexpr utx::ux32 y1 = std::byteswap(x1); // error, because utx::ux32 is a class.
		//constexpr utx::ux16 y2 = std::byteswap(x2); // error, because utx::ux16 is a class.

		constexpr utx::ux32 y1 = std::byteswap(x1()); // OK, call "operator()()"
		constexpr utx::ux16 y2 = std::byteswap(x2()); // OK, call "operator()()"

		utx::print(std::hex, std::showbase, y1, y2); // 0x78563412 0x3412

		/*
		These usage are OK too, have the same result:

		constexpr utx::ux32 y1 = std::byteswap((utx::u32)x1); // call "operator basic_type ()"
		constexpr utx::ux32 y2 = std::byteswap((utx::u16)x2); // call "operator basic_type ()"

		constexpr utx::ux32 y1 = std::byteswap(x1.value());
		constexpr utx::ux32 y2 = std::byteswap(x2.value());
		*/
	}
	{
		constexpr utx::ux32 x1 = 0x12345678;
		constexpr utx::ux16 x2 = 0x1234;
		constexpr utx::ux32 y1 = std::byteswap<utx::u32>(x1); // OK
		constexpr utx::ux16 y2 = std::byteswap<utx::u16>(x2);
		utx::print(std::hex, std::showbase, y1, y2); // 0x78563412 0x3412
	}
}
