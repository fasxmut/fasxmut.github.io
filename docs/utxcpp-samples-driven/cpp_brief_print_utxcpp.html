c++ brief print

c++ simple print using utxcpp. (utxcpp)

Print one value for any printable types

utx::print 

#include <utxcpp/core.hpp>

int main()
{
	utx::print(123);
	utx::print('N');
	utx::print(true);
	utx::string str = "Hello c++!";
	utx::print(str);
}

output: 

123
N
true
Hello c++!

Print many different types of values

utx::print 

#include <utxcpp/core.hpp>

int main()
{
	utx::print(123, 'N', true, "Hello c++!");
	utx::print(123, 'N', true, "Hello c++!");
}

(utx::string)

output: 

123 N true Hello c++!
123 N true Hello c++!

Print one STL range container

utx::print 

#include <utxcpp/core.hpp>

int main()
{
	std::vector<utx::ix32> vector{1,2,3,4,5};
	utx::print(vector);
	std::vector<utx::ux32> vector2{1,2,3};
	utx::print(vector2);
}

(utx::ix32) (utx::ux32)

output: 

1 2 3 4 5
1 2 3

Print many STL containers

utx::print 

#include <utxcpp/core.hpp>

int main()
{
	std::vector<utx::ix32> vector{1,2,3,4,5};
	std::vector<utx::ux32> vector2{1,2,3};
	utx::print(vector, vector2);
}

output:

1 2 3

Print one or many tuples

utx::print 

#include <utxcpp/core.hpp>

using namespace utx::string_literals;

int main()
{
	std::tuple t1{123, "Hello c++!", "Hello Utxcpp!"_us, true};
	utx::print("t1=>");
	utx::print(t1);

	utx::print("t1 t2=>");
	std::tuple t2{'R', false, 2.5};
	utx::print(t1, t2);

	std::vector<utx::ix32> v1{1, 2, 3};
	utx::print("t1 t2 v1=>");
	utx::print(t1, t2, v1);

	std::string str = "Hello c++!";
	utx::print("t1 t2 v1 str=>");
	utx::print(t1, t2, v1, str);
}

utx::string_literals
