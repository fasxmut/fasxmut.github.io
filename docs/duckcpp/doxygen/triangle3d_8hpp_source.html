68 nub::vector3d<T> closestPointOnTriangle(const nub::vector3d<T>& p) const
69 {
70 const nub::vector3d<T> rab = line3d<T>(pointA, pointB).getClosestPoint(p);
71 const nub::vector3d<T> rbc = line3d<T>(pointB, pointC).getClosestPoint(p);
72 const nub::vector3d<T> rca = line3d<T>(pointC, pointA).getClosestPoint(p);
73
74 const T d1 = rab.getDistanceFrom(p);
75 const T d2 = rbc.getDistanceFrom(p);
76 const T d3 = rca.getDistanceFrom(p);
77
78 if (d1 < d2)
79 return d1 < d3 ? rab : rca;
80
81 return d2 < d3 ? rbc : rca;
82 }

85 /*
86 \param p Point to test. Assumes that this point is already
87 on the plane of the triangle.
88 \return True if the point is inside the triangle, otherwise false. */
89 bool isPointInside(const vector3d<T>& p) const
90 {
91 vector3d<dcpp::float64_kt> a_Float64((dcpp::float64_kt)pointA.X, (dcpp::float64_kt)pointA.Y, (dcpp::float64_kt)pointA.Z);
92 vector3d<dcpp::float64_kt> b_Float64((dcpp::float64_kt)pointB.X, (dcpp::float64_kt)pointB.Y, (dcpp::float64_kt)pointB.Z);
93 vector3d<dcpp::float64_kt> c_Float64((dcpp::float64_kt)pointC.X, (dcpp::float64_kt)pointC.Y, (dcpp::float64_kt)pointC.Z);
94 vector3d<dcpp::float64_kt> p_Float64((dcpp::float64_kt)p.X, (dcpp::float64_kt)p.Y, (dcpp::float64_kt)p.Z);
95 return (isOnSameSide(p_Float64, a_Float64, b_Float64, c_Float64) &&
96 isOnSameSide(p_Float64, b_Float64, a_Float64, c_Float64) &&
97 isOnSameSide(p_Float64, c_Float64, a_Float64, b_Float64));
98 }

108 bool isPointInsideFast(const vector3d<T>& p) const
109 {
110 const vector3d<T> a = pointC - pointA;
111 const vector3d<T> b = pointB - pointA;
112 const vector3d<T> c = p - pointA;
113
114 const dcpp::float64_kt dotAA = a.dotProduct( a);
115 const dcpp::float64_kt dotAB = a.dotProduct( b);
116 const dcpp::float64_kt dotAC = a.dotProduct( c);
117 const dcpp::float64_kt dotBB = b.dotProduct( b);
118 const dcpp::float64_kt dotBC = b.dotProduct( c);
119
120 // get coordinates in barycentric coordinate system
121 const dcpp::float64_kt invDenom = 1/(dotAA * dotBB - dotAB * dotAB);
122 const dcpp::float64_kt u = (dotBB * dotAC - dotAB * dotBC) * invDenom;
123 const dcpp::float64_kt v = (dotAA * dotBC - dotAB * dotAC ) * invDenom;
124
125 // We count border-points as inside to keep downward compatibility.
126 // Rounding-error also needed for some test-cases.
127 return (u > -ROUNDING_ERROR_Float32) && (v >= 0) && (u + v < 1+ROUNDING_ERROR_Float32);
128
129 }

169 bool getIntersectionOfPlaneWithLine(const vector3d<T>& linePoint,
170 const vector3d<T>& lineVect, vector3d<T>& outIntersection) const
171 {
172 // Work with dcpp::float64_kt to get more precise results (makes enough difference to be worth the casts).
173 const vector3d<dcpp::float64_kt> linePoint_Float64(linePoint.X, linePoint.Y, linePoint.Z);
174 const vector3d<dcpp::float64_kt> lineVect_Float64(lineVect.X, lineVect.Y, lineVect.Z);
175 vector3d<dcpp::float64_kt> outIntersection_Float64;
176
177 nub::triangle3d<dcpp::float64_kt> triangle_Float64(vector3d<dcpp::float64_kt>((dcpp::float64_kt)pointA.X, (dcpp::float64_kt)pointA.Y, (dcpp::float64_kt)pointA.Z)
178 ,vector3d<dcpp::float64_kt>((dcpp::float64_kt)pointB.X, (dcpp::float64_kt)pointB.Y, (dcpp::float64_kt)pointB.Z)
179 , vector3d<dcpp::float64_kt>((dcpp::float64_kt)pointC.X, (dcpp::float64_kt)pointC.Y, (dcpp::float64_kt)pointC.Z));
180 const vector3d<dcpp::float64_kt> normal_Float64 = triangle_Float64.getNormal().normalize();
181 dcpp::float64_kt t2;
182
183 if ( nub::iszero ( t2 = normal_Float64.dotProduct(lineVect_Float64) ) )
184 return false;
185
186 dcpp::float64_kt d = triangle_Float64.pointA.dotProduct(normal_Float64);
187 dcpp::float64_kt t = -(normal_Float64.dotProduct(linePoint_Float64) - d) / t2;
188 outIntersection_Float64 = linePoint_Float64 + (lineVect_Float64 * t);
189
190 outIntersection.X = (T)outIntersection_Float64.X;
191 outIntersection.Y = (T)outIntersection_Float64.Y;
192 outIntersection.Z = (T)outIntersection_Float64.Z;
193 return true;
194 }

244 // Using dcpp::float64_kt instead of <T> to avoid integer overflows when T=int (maybe also less floating point troubles).
245 bool isOnSameSide(const vector3d<dcpp::float64_kt>& p1, const vector3d<dcpp::float64_kt>& p2,
246 const vector3d<dcpp::float64_kt>& a, const vector3d<dcpp::float64_kt>& b) const
247 {
248 vector3d<dcpp::float64_kt> bminusa = b - a;
249 vector3d<dcpp::float64_kt> cp1 = bminusa.crossProduct(p1 - a);
250 vector3d<dcpp::float64_kt> cp2 = bminusa.crossProduct(p2 - a);
251 dcpp::float64_kt res = cp1.dotProduct(cp2);
252 if ( res < 0 )
253 {
254 // This catches some floating point troubles.
255 // Unfortunately slightly expensive and we don't really know the best epsilon for iszero.
256 vector3d<dcpp::float64_kt> cp1n = bminusa.normalize().crossProduct((p1 - a).normalize());
257 if (nub::iszero(cp1n.X, (dcpp::float64_kt)ROUNDING_ERROR_Float32)
258 && nub::iszero(cp1n.Y, (dcpp::float64_kt)ROUNDING_ERROR_Float32)
259 && nub::iszero(cp1n.Z, (dcpp::float64_kt)ROUNDING_ERROR_Float32) )
260 {
261 res = 0.f;
262 }
263 }
264 return (res >= 0.0f);
265 }
