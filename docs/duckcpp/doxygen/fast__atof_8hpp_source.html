57inline dcpp::uint32_kt strtoul10(const char* in, const char** out=0)
58{
59 if (!in)
60 {
61 if (out)
62 *out = in;
63 return 0;
64 }
65
66 bool overflow=false;
67 dcpp::uint32_kt unsignedValue = 0;
68 while ( ( *in >= '0') && ( *in <= '9' ))
69 {
70 const dcpp::uint32_kt tmp = ( unsignedValue * 10 ) + ( *in - '0' );
71 if (tmp<unsignedValue)
72 {
73 unsignedValue=(dcpp::uint32_kt)0xffffffff;
74 overflow=true;
75 }
76 if (!overflow)
77 unsignedValue = tmp;
78 ++in;
79 }
80
81 if (out)
82 *out = in;
83
84 return unsignedValue;
85}

97inline dcpp::int32_kt strtol10(const char* in, const char** out=0)
98{
99 if (!in)
100 {
101 if (out)
102 *out = in;
103 return 0;
104 }
105
106 const bool negative = ('-' == *in);
107 if (negative || ('+' == *in))
108 ++in;
109
110 const dcpp::uint32_kt unsignedValue = strtoul10(in,out);
111 if (unsignedValue > (dcpp::uint32_kt)INT_MAX)
112 {
113 if (negative)
114 return (dcpp::int32_kt)INT_MIN;
115 else
116 return (dcpp::int32_kt)INT_MAX;
117 }
118 else
119 {
120 if (negative)
121 return -((dcpp::int32_kt)unsignedValue);
122 else
123 return (dcpp::int32_kt)unsignedValue;
124 }
125}

133inline dcpp::uint32_kt ctoul16(char in)
134{
135 if (in >= '0' && in <= '9')
136 return in - '0';
137 else if (in >= 'a' && in <= 'f')
138 return 10u + in - 'a';
139 else if (in >= 'A' && in <= 'F')
140 return 10u + in - 'A';
141 else
142 return 0xffffffff;
143}

154inline dcpp::uint32_kt strtoul16(const char* in, const char** out=0)
155{
156 if (!in)
157 {
158 if (out)
159 *out = in;
160 return 0;
161 }
162
163 bool overflow=false;
164 dcpp::uint32_kt unsignedValue = 0;
165 while (true)
166 {
167 dcpp::uint32_kt tmp = 0;
168 if ((*in >= '0') && (*in <= '9'))
169 tmp = (unsignedValue << 4u) + (*in - '0');
170 else if ((*in >= 'A') && (*in <= 'F'))
171 tmp = (unsignedValue << 4u) + (*in - 'A') + 10;
172 else if ((*in >= 'a') && (*in <= 'f'))
173 tmp = (unsignedValue << 4u) + (*in - 'a') + 10;
174 else
175 break;
176 if (tmp<unsignedValue)
177 {
178 unsignedValue=(dcpp::uint32_kt)INT_MAX;
179 overflow=true;
180 }
181 if (!overflow)
182 unsignedValue = tmp;
183 ++in;
184 }
185
186 if (out)
187 *out = in;
188
189 return unsignedValue;
190}

244inline dcpp::uint32_kt strtoul_prefix(const char* in, const char** out=0)
245{
246 if (!in)
247 {
248 if (out)
249 *out = in;
250 return 0;
251 }
252 if ('0'==in[0])
253 return ('x'==in[1] ? strtoul16(in+2,out) : strtoul8(in+1,out));
254 return strtoul10(in,out);

266inline dcpp::float32_kt strtof10(const char* in, const char** out = 0)
267{
268 if (!in)
269 {
270 if (out)
271 *out = in;
272 return 0.f;
273 }
274
275 const dcpp::uint32_kt MAX_SAFE_UINT32_VALUE = UINT_MAX / 10 - 10;
276 dcpp::uint32_kt intValue = 0;
277
278 // Use integer arithmetic for as long as possible, for speed
279 // and precision.
280 while ( ( *in >= '0') && ( *in <= '9' ) )
281 {
282 // If it looks like we're going to overflow, bail out
283 // now and start using floating point.
284 if (intValue >= MAX_SAFE_UINT32_VALUE)
285 break;
286
287 intValue = (intValue * 10) + (*in - '0');
288 ++in;
289 }
290
291 dcpp::float32_kt floatValue = (dcpp::float32_kt)intValue;
292
293 // If there are any digits left to parse, then we need to use
294 // floating point arithmetic from here.
295 while ( ( *in >= '0') && ( *in <= '9' ) )
296 {
297 floatValue = (floatValue * 10.f) + (dcpp::float32_kt)(*in - '0');
298 ++in;
299 if (floatValue > FLT_MAX) // Just give up.
300 break;
301 }
302
303 if (out)
304 *out = in;
305
306 return floatValue;
307}

317inline const char* fast_atof_move(const char* in, dcpp::float32_kt& result)
318{
319 // Please run the regression test when making any modifications to this function.
320
321 result = 0.f;
322 if (!in)
323 return 0;
324
325 const bool negative = ('-' == *in);
326 if (negative || ('+'==*in))
327 ++in;
328
329 dcpp::float32_kt value = strtof10(in, &in);
330
331 if ( LOCALE_DECIMAL_POINTS.findFirst(*in) >= 0 )
332 {
333 const char* afterDecimal = ++in;
334 const dcpp::float32_kt decimal = strtof10(in, &afterDecimal);
335 const size_t numDecimals = afterDecimal - in;
336 if (numDecimals < DCPP_ATOF_TABLE_SIZE)
337 {
338 value += decimal * fast_atof_table[numDecimals];
339 }
340 else
341 {
342 value += decimal * (dcpp::float32_kt)pow(10.f, -(float)numDecimals);
343 }
344 in = afterDecimal;
345 }
346
347 if ('e' == *in || 'E' == *in)
348 {
349 ++in;
350 // Assume that the exponent is a whole number.
351 // strtol10() will deal with both + and - signs,
352 // but calculate as dcpp::float32_kt to prevent overflow at FLT_MAX
353 // Using pow with float cast instead of powf as otherwise accuracy decreases.
354 value *= (dcpp::float32_kt)pow(10.f, (dcpp::float32_kt)strtol10(in, &in));
355 }
356
357 result = negative?-value:value;
358 return in;
359}
