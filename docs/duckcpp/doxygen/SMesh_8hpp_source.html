87 void recalculateBoundingBox()
88 {
89 bool hasMeshBufferBBox = false;
90 for (uint32_kt i=0; i<MeshBuffers.size(); ++i)
91 {
92 const nub::aabbox3df& bb = MeshBuffers[i]->getBoundingBox();
93 if ( !bb.isEmpty() )
94 {
95 if ( !hasMeshBufferBBox )
96 {
97 hasMeshBufferBBox = true;
98 BoundingBox = bb;
99 }
100 else
101 {
102 BoundingBox.addInternalBox(bb);
103 }
104
105 }
106 }
107
108 if ( !hasMeshBufferBBox )
109 BoundingBox.reset(0.0f, 0.0f, 0.0f);
110 }
