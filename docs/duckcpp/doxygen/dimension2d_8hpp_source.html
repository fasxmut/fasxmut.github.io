150 dimension2d<T> getOptimalSize(
151 bool requirePowerOfTwo=true,
152 bool requireSquare=false,
153 bool larger=true,
154 dcpp::uint32_kt maxValue = 0) const
155 {
156 dcpp::uint32_kt i=1;
157 dcpp::uint32_kt j=1;
158 if (requirePowerOfTwo)
159 {
160 while (i<(dcpp::uint32_kt)Width)
161 i<<=1;
162 if (!larger && i!=1 && i!=(dcpp::uint32_kt)Width)
163 i>>=1;
164 while (j<(dcpp::uint32_kt)Height)
165 j<<=1;
166 if (!larger && j!=1 && j!=(dcpp::uint32_kt)Height)
167 j>>=1;
168 }
169 else
170 {
171 i=(dcpp::uint32_kt)Width;
172 j=(dcpp::uint32_kt)Height;
173 }
174
175 if (requireSquare)
176 {
177 if ((larger && (i>j)) || (!larger && (i<j)))
178 j=i;
179 else
180 i=j;
181 }
182
183 if ( maxValue > 0 && i > maxValue)
184 i = maxValue;
185
186 if ( maxValue > 0 && j > maxValue)
187 j = maxValue;
188
189 return dimension2d<T>((T)i,(T)j);
190 }
