59static inline dcpp::uint32_kt locale_lower ( dcpp::uint32_kt x )
60{
61 switch ( locale_current )
62 {
63 case DCPP_LOCALE_GERMAN:
64 case DCPP_LOCALE_ANSI:
65 break;
66 }
67 // ansi
68 return x >= 'A' && x <= 'Z' ? x + 0x20 : x;
69}

72static inline dcpp::uint32_kt locale_upper ( dcpp::uint32_kt x )
73{
74 switch ( locale_current )
75 {
76 case DCPP_LOCALE_GERMAN:
77 case DCPP_LOCALE_ANSI:
78 break;
79 }
80
81 // ansi
82 return x >= 'a' && x <= 'z' ? x + ( 'A' - 'a' ) : x;
83}

145 explicit basic_string(int number)
146 : array(0), allocated(0), used(0)
147 {
148 // store if negative and make positive
149
150 bool negative = false;
151 if (number < 0)
152 {
153 number *= -1;
154 negative = true;
155 }
156
157 // temporary buffer for 16 numbers
158
159 dcpp::char_kt tmpbuf[16]={0};
160 dcpp::uint32_kt idx = 15;
161
162 // special case '0'
163
164 if (!number)
165 {
166 tmpbuf[14] = '0';
167 *this = &tmpbuf[14];
168 return;
169 }
170
171 // add numbers
172
173 while(number && idx)
174 {
175 --idx;
176 tmpbuf[idx] = (dcpp::char_kt)('0' + (number % 10));
177 number /= 10;
178 }
179
180 // add sign
181
182 if (negative)
183 {
184 --idx;
185 tmpbuf[idx] = '-';
186 }
187
188 *this = &tmpbuf[idx];
189 }

311 return;

315 array = allocator.allocate(used); // new T[used];

319
320 array[length] = 0;

343 self_type & operator=(const self_type & other)
344 {
345 if (this == &other)
346 return *this;
347

349 if (used>allocated)

352 allocated = used;

354 }

356 const T* p = other.data();
357 for (dcpp::uint32_kt i=0; i<used; ++i, ++p)
358 array[i] = *p;
359
360 return *this;

373 template <class B>
374 self_type & operator=(const B* const c)
375 {
376 if (!c)
377 {
378 if (!array)
379 {
380 array = allocator.allocate(1); //new T[1];
381 allocated = 1;
382 }
383 used = 1;
384 array[0] = 0x0;
385 return *this;
386 }
387
388 if ((void*)c == (void*)array)
389 return *this;
390
391 dcpp::uint32_kt len = 0;
392 const B* p = c;
393 do
394 {
395 ++len;
396 } while(*p++);
397
398 // we'll keep the old string for a while, because the new
399 // string could be a part of the current string.
400 T* oldArray = array;
401
402 used = len;
403 if (used>allocated)
404 {
405 allocated = used;
406 array = allocator.allocate(used); //new T[used];
407 }
408
409 for (dcpp::uint32_kt l = 0; l<len; ++l)
410 array[l] = (T)c[l];
411
412 if (oldArray != array)
413 allocator.deallocate(oldArray); // delete [] oldArray;
414
415 return *this;
416 }

676 self_type & append(const T* const other, dcpp::uint32_kt length=0xffffffff)
677 {
678 if (!other)
679 return *this;
680
681 dcpp::uint32_kt len = 0;
682 const T* p = other;
683 while(*p)
684 {
685 ++len;
686 ++p;
687 }
688 if (len > length)
689 len = length;
690
691 if (used + len > allocated)
692 reallocate(used + len);
693
694 --used;
695 ++len;
696
697 for (dcpp::uint32_kt l=0; l<len; ++l)
698 array[l+used] = *(other+l);
699
700 used += len;
701
702 return *this;
703 }

759 //\param pos Insert the characters before this index
760 //\param s String to insert. Must be at least of size n
761 //\param n Number of characters from string s to use.
762 self_type & insert(dcpp::uint32_kt pos, const char* s, dcpp::uint32_kt n)
763 {
764 if ( pos < used )
765 {
766 reserve(used+n);
767
768 // move stuff behind insert point
769 const dcpp::uint32_kt end = used+n-1;
770 for (dcpp::uint32_kt i=0; i<used-pos; ++i)
771 {
772 array[end-i] = array[end-(i+n)];
773 }
774 used += n;
775
776 for (dcpp::uint32_kt i=0; i<n; ++i)
777 {
778 array[pos+i] = s[i];
779 }
780 }
781
782 return *this;
783 }

940 template <class B>
941 dcpp::int32_kt find(const B* const str, const dcpp::uint32_kt start = 0) const
942 {
943 if (str && *str)
944 {
945 dcpp::uint32_kt len = 0;
946
947 while (str[len])
948 ++len;
949
950 if (len > used-1)
951 return -1;
952
953 for (dcpp::uint32_kt i=start; i<used-len; ++i)
954 {
955 dcpp::uint32_kt j=0;
956
957 while(str[j] && array[i+j] == str[j])
958 ++j;
959
960 if (!str[j])
961 return i;
962 }
963 }
964
965 return -1;
966 }

973 self_type subString(dcpp::uint32_kt begin, dcpp::int32_kt length, bool make_lower = false ) const
974 {
975 // clamp length to maximal value
976 if ((length+begin) > size())
977 length = size()-begin;
978 // if start after string
979 // or no proper substring length
980 if ((length <= 0) || (begin>=size()))
981 return self_type{""};
982
983 self_type o;
984 o.reserve(length+1);
985
986 if ( !make_lower )
987 {
988 for (dcpp::int32_kt i=0; i<length; ++i)
989 o.array[i] = array[i+begin];
990 }
991 else
992 {
993 for (dcpp::int32_kt i=0; i<length; ++i)
994 o.array[i] = locale_lower ( array[i+begin] );
995 }
996
997 o.array[length] = 0;
998 o.used = length + 1;
999
1000 return o;
1001 }

1100 self_type & replace(const self_type & toReplace, const self_type & replaceWith)
1101 {
1102 if (toReplace.size() == 0)
1103 return *this;
1104
1105 const T* other = toReplace.data();
1106 const T* replace = replaceWith.data();
1107 const dcpp::uint32_kt other_size = toReplace.size();
1108 const dcpp::uint32_kt replace_size = replaceWith.size();
1109
1110 // Determine the delta. The algorithm will change depending on the delta.
1111 dcpp::int32_kt delta = replace_size - other_size;
1112
1113 // A character for character replace. The string will not shrink or grow.
1114 if (delta == 0)
1115 {
1116 dcpp::int32_kt pos = 0;
1117 while ((pos = find(other, pos)) != -1)
1118 {
1119 for (dcpp::uint32_kt i = 0; i < replace_size; ++i)
1120 array[pos + i] = replace[i];
1121 ++pos;
1122 }
1123 return *this;
1124 }
1125
1126 // We are going to be removing some characters. The string will shrink.
1127 if (delta < 0)
1128 {
1129 dcpp::uint32_kt i = 0;
1130 for (dcpp::uint32_kt pos = 0; pos < used; ++i, ++pos)
1131 {
1132 // Is this potentially a match?
1133 if (array[pos] == *other)
1134 {
1135 // Check to see if we have a match.
1136 dcpp::uint32_kt j;
1137 for (j = 0; j < other_size; ++j)
1138 {
1139 if (array[pos + j] != other[j])
1140 break;
1141 }
1142
1143 // If we have a match, replace characters.
1144 if (j == other_size)
1145 {
1146 for (j = 0; j < replace_size; ++j)
1147 array[i + j] = replace[j];
1148 i += replace_size - 1;
1149 pos += other_size - 1;
1150 continue;
1151 }
1152 }
1153
1154 // No match found, just copy characters.
1155 array[i] = array[pos];
1156 }
1157 array[i-1] = 0;
1158 used = i;
1159
1160 return *this;
1161 }
1162
1163 // We are going to be adding characters, so the string size will increase.
1164 // Count the number of times toReplace exists in the string so we can allocate the new size.
1165 dcpp::uint32_kt find_count = 0;
1166 dcpp::int32_kt pos = 0;
1167 while ((pos = find(other, pos)) != -1)
1168 {
1169 ++find_count;
1170 ++pos;
1171 }
1172
1173 // Re-allocate the string now, if needed.
1174 dcpp::uint32_kt len = delta * find_count;
1175 if (used + len > allocated)
1176 reallocate(used + len);
1177
1178 // Start replacing.
1179 pos = 0;
1180 while ((pos = find(other, pos)) != -1)
1181 {
1182 T* start = array + pos + other_size - 1;
1183 T* ptr = array + used - 1;
1184 T* end = array + delta + used -1;
1185
1186 // Shift characters to make room for the string.
1187 while (ptr != start)
1188 {
1189 *end = *ptr;
1190 --ptr;
1191 --end;
1192 }
1193
1194 // Add the new string now.
1195 for (dcpp::uint32_kt i = 0; i < replace_size; ++i)
1196 array[pos + i] = replace[i];
1197
1198 pos += replace_size;
1199 used += delta;
1200 }
1201
1202 return *this;
1203 }
