386 class SJoystickEvent
387 {
388 public:
389 enum
390 {
391 NUMBER_OF_BUTTONS = 32,
392
393 AXIS_X = 0, // e.g. analog stick 1 left to right
394 AXIS_Y, // e.g. analog stick 1 top to bottom
395 AXIS_Z, // e.g. throttle, or analog 2 stick 2 left to right
396 AXIS_R, // e.g. rudder, or analog 2 stick 2 top to bottom
397 AXIS_U,
398 AXIS_V,
399 NUMBER_OF_AXES=18 // (please tell Duckcpp maintainers if you absolutely need more axes)
400 };
401
404 dcpp::uint32_kt ButtonStates;
405
414 dcpp::int16_kt Axis[NUMBER_OF_AXES];
415
421 dcpp::uint16_kt POV;
422
424
426 dcpp::uint8_kt Joystick;
427
429 bool IsButtonPressed(dcpp::uint32_kt button) const
430 {
431 if(button >= (dcpp::uint32_kt)NUMBER_OF_BUTTONS)
432 return false;
433
434 return (ButtonStates & (1 << button)) ? true : false;
435 }
