74 void reset()
75 {
76 EnableFlags = 0;
77 EnablePasses = 0;
78 Enabled = false;
79 for (dcpp::uint32_kt i = 0; i < MATERIAL_MAX_TEXTURES; ++i)
80 {
81 EnableLayerFlags[i] = true; // doesn't do anything unless EnableFlags is set, just saying by default all texture layers are affected by flags
82 EnableTextures[i] = false;
83 EnableLayers[i] = false;
84 }
85 MaterialTypes.clear();
86 }

89 void apply(SMaterial& material)
90 {
91 if (Enabled)
92 {
93 for (dcpp::uint32_kt i = 0; i < MaterialTypes.size(); ++i)
94 {
95 const SMaterialTypeReplacement& mtr = MaterialTypes[i];
96 if (mtr.Original < 0 || (dcpp::int32_kt)mtr.Original == material.MaterialType)
97 material.MaterialType = (E_MATERIAL_TYPE)mtr.Replacement;
98 }
99 for (dcpp::uint32_kt f=0; f<32; ++f)
100 {
101 const dcpp::uint32_kt num=(1<<f);
102 if (EnableFlags & num)
103 {
104 switch (num)
105 {
106 case EMF_WIREFRAME: material.Wireframe = Material.Wireframe; break;
107 case EMF_POINTCLOUD: material.PointCloud = Material.PointCloud; break;
108 case EMF_GOURAUD_SHADING: material.GouraudShading = Material.GouraudShading; break;
109 case EMF_LIGHTING: material.Lighting = Material.Lighting; break;
110 case EMF_ZBUFFER: material.ZBuffer = Material.ZBuffer; break;
111 case EMF_ZWRITE_ENABLE: material.ZWriteEnable = Material.ZWriteEnable; break;
112 case EMF_BACK_FACE_CULLING: material.BackfaceCulling = Material.BackfaceCulling; break;
113 case EMF_FRONT_FACE_CULLING: material.FrontfaceCulling = Material.FrontfaceCulling; break;
114 case EMF_BILINEAR_FILTER:
115 for ( dcpp::uint32_kt i=0; i<MATERIAL_MAX_TEXTURES; ++i)
116 {
117 if ( EnableLayerFlags[i] )
118 {
119 material.TextureLayer[i].BilinearFilter = Material.TextureLayer[i].BilinearFilter;
120 }
121 }
122 break;
123 case EMF_TRILINEAR_FILTER:
124 for ( dcpp::uint32_kt i=0; i<MATERIAL_MAX_TEXTURES; ++i)
125 {
126 if ( EnableLayerFlags[i] )
127 {
128 material.TextureLayer[i].TrilinearFilter = Material.TextureLayer[i].TrilinearFilter;
129 }
130 }
131 break;
132 case EMF_ANISOTROPIC_FILTER:
133 for ( dcpp::uint32_kt i=0; i<MATERIAL_MAX_TEXTURES; ++i)
134 {
135 if ( EnableLayerFlags[i] )
136 {
137 material.TextureLayer[i].AnisotropicFilter = Material.TextureLayer[i].AnisotropicFilter;
138 }
139 }
140 break;
141 case EMF_FOG_ENABLE: material.FogEnable = Material.FogEnable; break;
142 case EMF_NORMALIZE_NORMALS: material.NormalizeNormals = Material.NormalizeNormals; break;
143 case EMF_TEXTURE_WRAP:
144 for ( dcpp::uint32_kt i=0; i<MATERIAL_MAX_TEXTURES; ++i)
145 {
146 if ( EnableLayerFlags[i] )
147 {
148 material.TextureLayer[i].TextureWrapU = Material.TextureLayer[i].TextureWrapU;
149 material.TextureLayer[i].TextureWrapV = Material.TextureLayer[i].TextureWrapV;
150 material.TextureLayer[i].TextureWrapW = Material.TextureLayer[i].TextureWrapW;
151 }
152 }
153 break;
154 case EMF_ANTI_ALIASING: material.AntiAliasing = Material.AntiAliasing; break;
155 case EMF_COLOR_MASK: material.ColorMask = Material.ColorMask; break;
156 case EMF_COLOR_MATERIAL: material.ColorMaterial = Material.ColorMaterial; break;
157 case EMF_USE_MIP_MAPS: material.UseMipMaps = Material.UseMipMaps; break;
158 case EMF_BLEND_OPERATION: material.BlendOperation = Material.BlendOperation; break;
159 case EMF_BLEND_FACTOR: material.BlendFactor = Material.BlendFactor; break;
160 case EMF_POLYGON_OFFSET:
161 material.PolygonOffsetDirection = Material.PolygonOffsetDirection;
162 material.PolygonOffsetFactor = Material.PolygonOffsetFactor;
163 material.PolygonOffsetDepthBias = Material.PolygonOffsetDepthBias;
164 material.PolygonOffsetSlopeScale = Material.PolygonOffsetSlopeScale;
165 break;
166 }
167 }
168 }
169 for(dcpp::uint32_kt i=0; i<MATERIAL_MAX_TEXTURES; ++i )
170 {
171 if ( EnableLayers[i] )
172 {
173 material.TextureLayer[i] = Material.TextureLayer[i];
174 }
175 else if ( EnableTextures[i] )
176 {
177 material.TextureLayer[i].Texture = Material.TextureLayer[i].Texture;
178 }
179 }
180 }
181 }
182 };
