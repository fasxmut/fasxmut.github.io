113 virtual void setType(dcpp::video::E_INDEX_TYPE indexType) override
114 {
115 if ( Indices && Indices->getType() == indexType )
116 return;
117
118 IIndexList *NewIndices=0;
119
120 switch (indexType)
121 {
122 case dcpp::video::EIT_16BIT:
123 {
124 NewIndices=new PSpecificIndexList<dcpp::uint16_kt>;
125 break;
126 }
127 case dcpp::video::EIT_32BIT:
128 {
129 NewIndices=new PSpecificIndexList<dcpp::uint32_kt>;
130 break;
131 }
132 }
133
134 if (Indices)
135 {
136 NewIndices->reallocate( Indices->size() );
137
138 for(dcpp::uint32_kt n=0;n<Indices->size();++n)
139 NewIndices->push_back((*Indices)[n]);
140
141 delete Indices;
142 }
143
144 Indices=NewIndices;
145 }
