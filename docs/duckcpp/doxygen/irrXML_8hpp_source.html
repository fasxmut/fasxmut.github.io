441 xmlChar(wchar_t in) : c(static_cast<T>(in)) {}
442#if defined(__BORLANDC__)
443 // Note - removing explicit for Borland was to get it to even compile.
444 // There haven't been any kind of tests for that besides that.
445 xmlChar(unsigned char in) : c(static_cast<T>(in)) {}
446 xmlChar(unsigned short in) : c(static_cast<T>(in)) {}
447 xmlChar(unsigned int in) : c(static_cast<T>(in)) {}
448 xmlChar(unsigned long in) : c(static_cast<T>(in)) {}
449#else
450 explicit xmlChar(unsigned char in) : c(static_cast<T>(in)) {}
451

453 explicit xmlChar(unsigned short in) : c(static_cast<T>(in)) {}
454#endif
455 explicit xmlChar(unsigned int in) : c(static_cast<T>(in)) {}
456 explicit xmlChar(unsigned long in) : c(static_cast<T>(in)) {}
457#endif
458 operator T() const { return c; }
