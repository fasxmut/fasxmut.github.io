utx::print cpp Cartesian Product

utx::print can print std::views::cartesian_product directly.

Main Program 

#include <ranges>
#include <utxcpp/core.hpp>
#include <vector>
#include <list>

using namespace std::string_literals;

int main()
{
	auto v1 = std::vector{2,3};
	const auto & v2 = std::list{"c++"s, "Hello!"s};
	const auto v3 = {true, false};
	utx::print("Cartesian product:", std::views::cartesian_product(v1, v2, v3));
}
/*
Cartesian product:
2 c++ true
2 c++ false
2 Hello! true
2 Hello! false
3 c++ true
3 c++ false
3 Hello! true
3 Hello! false
*/
