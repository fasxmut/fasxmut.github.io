c++ slerp and quaternion

Create quaternion 

nirt::core::quaternion q1;
nirt::core::quaternion q2{1, 2, 3};	// x, y, z
nirt::core::quaternion q3{1,2,3,4};	// x, y, z, w

Access X, Y, Z, W 

std::cout << q3.X << ',' << q3.Y << ',' << q3.Z << ',' << q3.W << std::endl;

quaternion slerp 

q1.slerp(q2, q3, 0.7);	// set q1 to the slerp of q2 and q3, with time 0.7

Main Program 

#include <nirtcpp.hpp>
#include <utxcpp/core.hpp>

namespace nirt::core
{

std::ostream & operator<<(std::ostream & out, const nirt::core::quaternion & q)
{
	return out << '{' << q.X << ',' << q.Y << ',' << q.Z << ',' << q.W << '}';
}

}	//

int main()
{
	nirt::core::quaternion
		q1,
		q2{1,2,3},
		q3{1,2,3,4}
	;
	
	utx::print.set_value_delim('\n');
	utx::print("q1 q2 q3=>", q1, q2, q3, '\n');
	utx::print.restore_format();
	
	q1.slerp(q2, q3, 0.45);
	utx::print("q1=>", q1);
	
	q1.slerp(q2, q3, 0.75);
	utx::print("q1=>", q1);
}
