BEntry Rename, Haiku c++ 

#include <Entry.h>
#include <iostream>

int main()
{
	BEntry file{"audio.wav"};

	std::cout << file.Name() << " exists: " << std::boolalpha << file.Exists() << std::endl;
	
	// entry_ref e_ref;
	// file.GetRef(&e_ref);

	if (file.Exists())
	{
		std::cout << "Rename audio.wav to music.wav, sure? (y/n) " << std::flush;
		std::string ans;
		std::getline(std::cin, ans);
		std::cout << std::endl;
		if (ans == "y")
		{
			file.Rename("music.wav");
			std::cout << "Renamed OK." << std::endl;
		}
	}
	else
		std::cout << "not exists!" << std::endl;


	// file.Remove();

	std::cout << std::endl;
}
