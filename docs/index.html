#include <algorithm>

int main()
{
  // c++ array
  char src[] = {"Hello, c++!"};
  // Note: src[11] == '\0'
  int size = 11 + 1;
  char * dst = new char[size];
  std::copy(src, src+size, dst);
  delete [] dst;
}
