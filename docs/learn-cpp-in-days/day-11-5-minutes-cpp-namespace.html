Make your own namespace 

namespace my_space
{
	float x = 2.5f;
}

std::cout << my_space::x << std::endl;

Make your own nested namespace 1 

namespace my_space::my_name
{
	float x = 5.2f;
}

std::cout << my_space::my_name::x << std::endl;

Make your own nested namespace 2 

namespace my_space
{
	namespace my_star
	{
		float x = 2.5f;
	}
}

std::cout << my_space::my_star::x << std::endl;
