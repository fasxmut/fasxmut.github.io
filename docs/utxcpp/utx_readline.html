Example 

#include <utxcpp/core.hpp>

int main() {
	utx::print(
		"Your Name is:",
		utx::readline("Please Input Your Name: ").read(),
		"\nHow are you."
	);
}
Result: 
Please Input Your Name: Fas Xmut
Your Name is: Fas Xmut
How are you.
