utfs::wkdir

utfs::wkdir returns current working directory path. 

namespace utfs
{

utfs::path wkdir()
{
	...
}

}

utfs::wkdir c++ example 

#include <utxcpp/utfs.hpp>
#include <utxcpp/core.hpp>

int main()
{
	utx::print(utfs::wkdir());
}
