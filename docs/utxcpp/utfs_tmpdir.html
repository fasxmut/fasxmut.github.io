utfs::tmpdir

utfs::tmpdir

utfs::tmpdir returns Temporary Directory Path, temp directory path is defined by os, Unix: /tmp; Windows: "C:\Windows\Temp" for example. 

namespace utfs
{

utfs::path tmpdir()
{
	...
}

}

utfs::tmpdir c++ example 

#include <utxcpp/utfs.hpp>
#include <utxcpp/core.hpp>

int main()
{
	utx::print(utfs::tmpdir());
}
