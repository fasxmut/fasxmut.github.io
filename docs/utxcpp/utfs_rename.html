utfs::rename

utfs::rename changes the name of a path to a new name.

{

utfs::rename c++ example 

#include <utxcpp/utfs.hpp>
#include <utxcpp/core.hpp>

int main()
{
	utfs::mkdir("oldname");
	utx::print(utfs::exists("newname")); // false
	utfs::rename("oldname", "newname");
	utx::print(utfs::exists("newname")); // true
}
