Example 

#include <utxcpp/flat.hpp>

int main()
{
	//uflat::vector<0, utx::i32> v1; // error: not satisfied. Requires at least 1.
	constexpr uflat::vector<1, utx::i32> v2;
	static_assert(v2[0] == 0);

	constexpr uflat::vector<3, utx::i32> v3{2,3};
	static_assert(v3[0] == 2);
	static_assert(v3[1] == 3);
	static_assert(v3.w() == v3.z());
	static_assert(v3.z() == 0);
	static_assert(v3.y() == 3);
	static_assert(v3.x() == 2);

	constexpr uflat::vector<5, utx::i32> v4{3,4,5,6,7,8,9};
	static_assert(v4.size() == 5);
	static_assert(v4[0] == 3);
	static_assert(v4[1] == 4);
	static_assert(v4[2] == 5);
	static_assert(v4[3] == 6);
	static_assert(v4[4] == 7);

	static_assert(v4.w() == 6);
	static_assert(v4.z() == 5);
	static_assert(v4.y() == 4);
	static_assert(v4.x() == 3);
}
