uflat::initializer_list

class 

template <std::integral auto __size, typename type1>
	requires (__size >= 0)
class initializer_list final;

constexpr support.

uflat::initializer_list : 

constexpr uflat::initializer_list<7, utx::i32> list = {1,2,3,4,5,6,7};

It is not an error if list size doesn't equal to template size:

  • list size > template size: The rest of values in list will be ignored.
  • list size < template size: The rest of stored values will be set to 0.

Example 

#include <utxcpp/flat.hpp>

int main()
{
	constexpr uflat::initializer_list<0, utx::i32> list1;
	static_assert(list1.size() == 0);

	constexpr uflat::initializer_list<1, utx::i32> list2;
	static_assert(list2.size() == 1);

	constexpr uflat::initializer_list<2, utx::i32> list3;
	static_assert(list3.size() == 2);

	constexpr uflat::initializer_list<3, utx::i32> list4{3,4};
	static_assert(list4[0] == 3);
	static_assert(list4[1] == 4);
	static_assert(list4[2] == 0);
}
