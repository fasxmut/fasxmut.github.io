utfs::path_literals

""_path

utfs::path_literals::operator""_path

utfs::path_literals::operator""_path

""_path is the utfs file system path literal of utxcpp. It forms an utfs::path object from a literal string.

""_path

class: utfs::path 

"a-string"_path

c++ example 

#include <utxcpp/utfs.hpp>

using namespace utfs::path_literals;

int main()
{
	if ("hell"_path.exists())
		utfs::remove("hell");
}
