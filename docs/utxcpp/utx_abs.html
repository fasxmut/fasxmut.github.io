utx::abs

template <utx::real_meric T>
inline constexpr utx::remove_cvref_t<T> abs(const T & value)

If value < 0, return -value; otherwise return value.

Header 

#include <utxcpp/math.hpp>

Example 

#include <utxcpp/core.hpp>

int main()
{
	for (utx::i32 i=-3; i<=3; i++)
		utx::printnl(utx::abs(i), ' ');
	utx::print();
}
