Example 

#include <esvcpp/core.hpp>

int main() {
	esv::print(
		"Your Name is:",
		esv::readline("Please Input Your Name: ").read(),
		"\nHow are you."
	);
}
Result: 
Please Input Your Name: Fas Xmut
Your Name is: Fas Xmut
How are you.
